public void calculate(View view) {
Calendar calendar = Calendar.getInstance();

int day = calendar.get(Calendar.DAY_OF_WEEK);
int hour = calendar.get(Calendar.HOUR_OF_DAY);
int minute = calendar.get(Calendar.MINUTE);
int second = calendar.get(Calendar.SECOND);

int realDate = 330;

if (day == 1) {
realDate = 6;
} else if (day == 2) {
realDate = 0;
} else if (day == 3) {
realDate = 1;
} else if (day == 4) {
realDate = 2;
} else if (day == 5) {
realDate = 3;
} else if (day == 6) {
realDate = 4;
} else if (day == 7) {
realDate = 5;
}
int hours = (realDate * 24) + hour;
int minutes = (hours * 60) + minute;
int seconds = (minutes * 60) + second;

DecimalFormat df5 = new DecimalFormat(".#####");
if (seconds <= 432000) {
double originalpercent = 0.00023148148148148;
double percent = originalpercent * seconds;
TextView doublePercent = (TextView) findViewById(R.id.random);
String stringPercent = String.valueOf(df5.format(percent));
doublePercent.setText(stringPercent);
} else {
int textHundred = 100;
TextView hundred = (TextView) findViewById(R.id.random);
String stringHundred = String.valueOf(textHundred);
hundred.setText(stringHundred);
}
} }
